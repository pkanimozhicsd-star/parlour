LIVE SAVE — save form data to ResponseFile.json (GitHub Pages cannot run PHP)
=============================================================================

FormSubmit only sends EMAIL. It does NOT save ResponseFile.json.
Use ONE of these options:

OPTION A — Free PHP host (same as local MAMP, easiest)
------------------------------------------------------
1. Sign up at https://www.infinityfree.com (free)
2. Create a site → upload save.php and ResponseFile.json to htdocs
3. Copy your site URL, e.g. https://deepzparlour.infinityfreeapp.com/save.php
4. In content.js set:
     "livePhpUrl": "https://deepzparlour.infinityfreeapp.com/save.php"
5. In save.php add your InfinityFree URL to allowedOrigins if needed
6. Push content.js to GitHub

OPTION B — Google Apps Script (saves ResponseFile.json in GitHub repo)
------------------------------------------------------------------------
1. Open https://script.google.com → New project
2. Paste the CODE below → Save
3. Project Settings → Script properties → Add:
   - GITHUB_TOKEN  = GitHub Personal Access Token (repo scope)
   - GITHUB_OWNER  = pkanimozhicsd-star
   - GITHUB_REPO   = parlour
   - GITHUB_PATH   = ResponseFile.json
4. Deploy → New deployment → Type: Web app
   - Execute as: Me
   - Who has access: Anyone
5. Copy the Web app URL
6. In content.js set:
     "googleScriptUrl": "https://script.google.com/macros/s/YOUR_ID/exec"
7. Push content.js to GitHub

GITHUB TOKEN: GitHub → Settings → Developer settings → Personal access tokens
Scope: repo (write access to parlour repository)

CODE TO PASTE IN GOOGLE APPS SCRIPT
-----------------------------------

function doPost(e) {
  try {
    var input = JSON.parse(e.postData.contents);
    var name = (input.name || '').trim();
    var mobile = (input.mobile || '').trim();
    var email = (input.email || '').trim();
    var service = (input.service || '').trim();

    if (!name || !mobile || !email || !service) {
      return jsonResponse({ success: false, message: 'All fields are required' });
    }
    if (!/^[0-9]{10}$/.test(mobile)) {
      return jsonResponse({ success: false, message: 'Mobile number must be 10 digits' });
    }

    var props = PropertiesService.getScriptProperties();
    var token = props.getProperty('GITHUB_TOKEN');
    var owner = props.getProperty('GITHUB_OWNER');
    var repo = props.getProperty('GITHUB_REPO');
    var path = props.getProperty('GITHUB_PATH');

    if (!token || !owner || !repo || !path) {
      return jsonResponse({ success: false, message: 'Script properties not configured' });
    }

    var apiUrl = 'https://api.github.com/repos/' + owner + '/' + repo + '/contents/' + path;
    var sha = null;
    var responses = [];

    var getRes = UrlFetchApp.fetch(apiUrl, {
      method: 'get',
      headers: { Authorization: 'Bearer ' + token, Accept: 'application/vnd.github+json' },
      muteHttpExceptions: true
    });

    if (getRes.getResponseCode() === 200) {
      var fileMeta = JSON.parse(getRes.getContentText());
      sha = fileMeta.sha;
      var decoded = Utilities.newBlob(Utilities.base64Decode(fileMeta.content)).getDataAsString();
      var parsed = JSON.parse(decoded);
      if (Array.isArray(parsed)) responses = parsed;
    } else if (getRes.getResponseCode() !== 404) {
      return jsonResponse({ success: false, message: 'Could not read ResponseFile.json from GitHub' });
    }

    var now = new Date();
    responses.push({
      submitted_at: Utilities.formatDate(now, 'Asia/Kolkata', 'dd/MM/yyyy, HH:mm:ss'),
      name: name,
      mobile: mobile,
      email: email,
      service: service
    });

    var body = {
      message: 'Add quote request from ' + name,
      content: Utilities.base64Encode(JSON.stringify(responses, null, 2)),
      sha: sha
    };

    var putRes = UrlFetchApp.fetch(apiUrl, {
      method: 'put',
      headers: {
        Authorization: 'Bearer ' + token,
        Accept: 'application/vnd.github+json',
        'Content-Type': 'application/json'
      },
      payload: JSON.stringify(body),
      muteHttpExceptions: true
    });

    if (putRes.getResponseCode() >= 200 && putRes.getResponseCode() < 300) {
      return jsonResponse({ success: true, message: 'Saved to ResponseFile.json', file: 'ResponseFile.json' });
    }

    return jsonResponse({ success: false, message: 'Could not write to GitHub: ' + putRes.getContentText() });
  } catch (err) {
    return jsonResponse({ success: false, message: String(err) });
  }
}

function jsonResponse(obj) {
  var out = ContentService.createTextOutput(JSON.stringify(obj));
  out.setMimeType(ContentService.MimeType.JSON);
  return out;
}
